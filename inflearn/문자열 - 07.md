import java.util.Scanner;

public class Main {

    public String solution(String s) {
        int lt = 0;
        int rt = s.length() - 1;

        s = s.toLowerCase();

        while (lt < rt) {
            if (s.charAt(lt) == s.charAt(rt)) {
                lt++;
                rt--;
            } else {
                return "NO";
            }
        }

        return "YES";
    }

    public static void main(String[] args) {
        Main T = new Main();
        Scanner kb = new Scanner(System.in);
        String next = kb.next();

        System.out.println(T.solution(next));;
    }
}
