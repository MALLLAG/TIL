import java.util.Scanner;

public class Main {

    public String solution(String s) {
        int lt = 0;
        int rt = s.length() - 1;

        char[] toCharArray = s.toCharArray();

        while (lt < rt) {
            if (!Character.isAlphabetic(toCharArray[lt])) {
                lt++;
            } else if (!Character.isAlphabetic(toCharArray[rt])) {
                rt--;
            } else {
                char c = toCharArray[lt];
                toCharArray[lt] = toCharArray[rt];
                toCharArray[rt] = c;

                lt++;
                rt--;
            }
        }

        
        return String.valueOf(toCharArray);
    }

    public static void main(String[] args) {
        Main T = new Main();
        Scanner kb = new Scanner(System.in);
        String next = kb.next();

        System.out.println(T.solution(next));

    }
}
